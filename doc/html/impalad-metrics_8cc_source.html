121 void ImpaladMetrics::CreateMetrics(MetricGroup* m) {
122  // Initialize impalad metrics
123  IMPALA_SERVER_START_TIME = m->AddProperty<string>(
124  ImpaladMetricKeys::IMPALA_SERVER_START_TIME, "");
125  IMPALA_SERVER_VERSION = m->AddProperty<string>(
126  ImpaladMetricKeys::IMPALA_SERVER_VERSION, GetVersionString(true));
127  IMPALA_SERVER_READY = m->AddProperty<bool>(
128  ImpaladMetricKeys::IMPALA_SERVER_READY, false);
129 
130  IMPALA_SERVER_NUM_QUERIES = m->AddCounter(
131  ImpaladMetricKeys::IMPALA_SERVER_NUM_QUERIES, 0L);
132  NUM_QUERIES_EXPIRED = m->AddCounter(
133  ImpaladMetricKeys::NUM_QUERIES_EXPIRED, 0L);
134  NUM_QUERIES_SPILLED = m->AddCounter(
135  ImpaladMetricKeys::NUM_QUERIES_SPILLED, 0L);
136  IMPALA_SERVER_NUM_FRAGMENTS = m->AddCounter(
137  ImpaladMetricKeys::IMPALA_SERVER_NUM_FRAGMENTS, 0L);
138  IMPALA_SERVER_NUM_OPEN_HS2_SESSIONS = m->AddGauge<int64_t>(
139  ImpaladMetricKeys::IMPALA_SERVER_NUM_OPEN_HS2_SESSIONS, 0L);
140  IMPALA_SERVER_NUM_OPEN_BEESWAX_SESSIONS = m->AddGauge<int64_t>(
141  ImpaladMetricKeys::IMPALA_SERVER_NUM_OPEN_BEESWAX_SESSIONS, 0L);
142  NUM_SESSIONS_EXPIRED = m->AddCounter(
143  ImpaladMetricKeys::NUM_SESSIONS_EXPIRED, 0L);
144  RESULTSET_CACHE_TOTAL_NUM_ROWS = m->AddGauge(
145  ImpaladMetricKeys::RESULTSET_CACHE_TOTAL_NUM_ROWS, 0L);
146  RESULTSET_CACHE_TOTAL_BYTES = m->AddGauge(
147  ImpaladMetricKeys::RESULTSET_CACHE_TOTAL_BYTES, 0L);
148 
149  // Initialize scan node metrics
150  NUM_RANGES_PROCESSED = m->AddCounter(
151  ImpaladMetricKeys::TOTAL_SCAN_RANGES_PROCESSED, 0L);
152  NUM_RANGES_MISSING_VOLUME_ID = m->AddCounter(
153  ImpaladMetricKeys::NUM_SCAN_RANGES_MISSING_VOLUME_ID, 0L);
154 
155  // Initialize memory usage metrics
156  MEM_POOL_TOTAL_BYTES = m->AddGauge<int64_t>(
157  ImpaladMetricKeys::MEM_POOL_TOTAL_BYTES, 0L, TUnit::BYTES);
158  HASH_TABLE_TOTAL_BYTES = m->AddGauge(
159  ImpaladMetricKeys::HASH_TABLE_TOTAL_BYTES, 0L, TUnit::BYTES);
160 
161  // Initialize insert metrics
162  NUM_FILES_OPEN_FOR_INSERT = m->AddGauge<int64_t>(
163  ImpaladMetricKeys::NUM_FILES_OPEN_FOR_INSERT, 0L);
164 
165  // Initialize IO mgr metrics
166  IO_MGR_NUM_OPEN_FILES = m->AddGauge<int64_t>(
167  ImpaladMetricKeys::IO_MGR_NUM_OPEN_FILES, 0L);
168  IO_MGR_NUM_BUFFERS = m->AddGauge<int64_t>(
169  ImpaladMetricKeys::IO_MGR_NUM_BUFFERS, 0L);
170  IO_MGR_TOTAL_BYTES = m->AddGauge<int64_t>(
171  ImpaladMetricKeys::IO_MGR_TOTAL_BYTES, 0L, TUnit::BYTES);
172  IO_MGR_NUM_UNUSED_BUFFERS = m->AddGauge<int64_t>(
173  ImpaladMetricKeys::IO_MGR_NUM_UNUSED_BUFFERS, 0L);
174 
175  IO_MGR_BYTES_READ = m->AddGauge(
176  ImpaladMetricKeys::IO_MGR_BYTES_READ, 0L, TUnit::BYTES);
177  IO_MGR_LOCAL_BYTES_READ = m->AddGauge(
178  ImpaladMetricKeys::IO_MGR_LOCAL_BYTES_READ, 0L, TUnit::BYTES);
179  IO_MGR_CACHED_BYTES_READ = m->AddGauge<int64_t>(
180  ImpaladMetricKeys::IO_MGR_CACHED_BYTES_READ, 0L, TUnit::BYTES);
181  IO_MGR_SHORT_CIRCUIT_BYTES_READ = m->AddGauge<int64_t>(
182  ImpaladMetricKeys::IO_MGR_SHORT_CIRCUIT_BYTES_READ, 0L, TUnit::BYTES);
183  IO_MGR_BYTES_WRITTEN = m->AddGauge<int64_t>(
184  ImpaladMetricKeys::IO_MGR_BYTES_WRITTEN, 0L, TUnit::BYTES);
185 
186  // Initialize catalog metrics
187  CATALOG_NUM_DBS = m->AddGauge<int64_t>(
188  ImpaladMetricKeys::CATALOG_NUM_DBS, 0L);
189  CATALOG_NUM_TABLES = m->AddGauge<int64_t>(
190  ImpaladMetricKeys::CATALOG_NUM_TABLES, 0L);
191  CATALOG_READY = m->AddProperty<bool>(
192  ImpaladMetricKeys::CATALOG_READY, false);
193 }
