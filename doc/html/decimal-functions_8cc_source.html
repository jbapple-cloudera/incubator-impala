35 DecimalVal DecimalFunctions::Abs(FunctionContext* context, const DecimalVal& val) {
36  if (val.is_null) return DecimalVal::null();
37  ColumnType type = AnyValUtil::TypeDescToColumnType(*context->GetArgType(0));
38  switch (type.GetByteSize()) {

40  return DecimalVal(abs(val.val4));

43  case 16:
44  return DecimalVal(abs(val.val16));
